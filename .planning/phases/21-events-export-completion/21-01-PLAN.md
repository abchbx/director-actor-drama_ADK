    /**
     * ★ D-21-05/D-21-06: 导出剧本为 Markdown，通过 Share Intent 分享
     */
    fun exportDrama() {
        viewModelScope.launch {
            _uiState.update { it.copy(isExporting = true) }
            val result = dramaRepository.exportDrama("markdown")
            _uiState.update { it.copy(isExporting = false) }
            result.onSuccess { response ->
                if (response.content.isNotBlank()) {
                    _events.emit(DramaDetailEvent.ShareExport(
                        content = response.content,
                        title = "剧本导出",
                    ))
                } else {
                    _events.emit(DramaDetailEvent.ShowSnackbar(
                        response.message.ifBlank { "导出成功但内容为空" }
                    ))
                }
            }.onFailure { e ->
                _events.emit(DramaDetailEvent.ShowSnackbar(
                    "导出失败: ${e.message ?: "未知错误"}"
                ))
            }
        }
    }
