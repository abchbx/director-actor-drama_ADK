    ```kotlin
    package com.drama.app.ui.screens.dramadetail.orchestrator

    import kotlinx.coroutines.CoroutineScope
    import kotlinx.coroutines.Job
    import kotlinx.coroutines.flow.MutableSharedFlow
    import kotlinx.coroutines.flow.MutableStateFlow
    import kotlinx.coroutines.flow.SharedFlow
    import kotlinx.coroutines.flow.StateFlow
    import kotlinx.coroutines.flow.asStateFlow
    import kotlinx.coroutines.isActive
    import kotlinx.coroutines.launch
    import javax.inject.Inject

    /**
     * WS 连接编排子组件（per D-23-01, D-23-02）。
     * 管理连接/重连/心跳/轮询降级，通过 SharedFlow 上报连接事件给主 VM。
     * 暴露 isWsConnected 状态供 BubbleMerger 实现数据源策略（per ARCH-10）。
     *
     * 架构选择：使用 @Inject constructor 而非 ViewModelSubComponent 基类。
     * 理由：@Inject 使子组件可独立测试（直接 new），无需 Hilt 测试运行器；
     * cleanup() 由主 VM onCleared 显式调用，等价于 ViewModelSubComponent.onCleared()。
     */
    class ConnectionOrchestrator @Inject constructor(
        private val webSocketManager: WebSocketManager,
        private val dramaRepository: DramaRepository
    ) {
        // D-23-02: 子组件通信用 SharedFlow 事件上报
        sealed class ConnectionEvent {
            data class Connected(val dramaId: String) : ConnectionEvent()
            data class Disconnected(val reason: String) : ConnectionEvent()
            data class Error(val message: String) : ConnectionEvent()
            data class Reconnecting(val attempt: Int) : ConnectionEvent()
            data class EventReceived(val event: WsEventDto) : ConnectionEvent()
        }

        private val _events = MutableSharedFlow<ConnectionEvent>(extraBufferCapacity = 64)
        val events: SharedFlow<ConnectionEvent> = _events

        // ARCH-10: 暴露 WS 连接状态，供 BubbleMerger 判断数据源
        private val _isWsConnected = MutableStateFlow(false)
        val isWsConnected: StateFlow<Boolean> = _isWsConnected.asStateFlow()

        private var connectJob: Job? = null
        private var reconnectJob: Job? = null
        private var replyPollJob: Job? = null
        private var connectGeneration: Int = 0

        /** 连接 WS，失败时降级为 REST 轮询 */
        fun connect(dramaId: String, token: String, scope: CoroutineScope) {
            disconnect()
            connectGeneration++
            val thisGeneration = connectGeneration
            connectJob = scope.launch {
                try {
                    webSocketManager.connect(dramaId, token)
                    _isWsConnected.value = true
                    _events.tryEmit(ConnectionEvent.Connected(dramaId))
                } catch (e: Exception) {
                    // D-23-16: WS 优先，失败降级 REST
                    _isWsConnected.value = false
                    _events.tryEmit(ConnectionEvent.Error("WS 连接失败，降级为 REST 轮询: ${e.message}"))
                    startReplyPolling(dramaId, scope)
                }
            }
        }

        /** 断开连接 */
        fun disconnect() {
            connectGeneration++
            connectJob?.cancel()
            reconnectJob?.cancel()
            replyPollJob?.cancel()
            webSocketManager.disconnect()
            _isWsConnected.value = false
        }

        /** REST 轮询降级 */
        private fun startReplyPolling(dramaId: String, scope: CoroutineScope) {
            replyPollJob?.cancel()
            replyPollJob = scope.launch {
                while (isActive) {
                    kotlinx.coroutines.delay(3000L)
                    try {
                        val result = dramaRepository.getDramaStatus(dramaId)
                        // 上报轮询事件
                    } catch (e: Exception) {
                        _events.tryEmit(ConnectionEvent.Error("轮询失败: ${e.message}"))
                    }
                }
            }
        }

        /** D-23-04: 主 VM onCleared 时调用统一清理 */
        fun cleanup() {
            disconnect()
        }
    }
    ```

    **2. BubbleMerger.kt (~200行) — 气泡列表管理/去重/线程安全 ID + ARCH-10 数据源策略**

    ```kotlin
    package com.drama.app.ui.screens.dramadetail.orchestrator

    import java.util.concurrent.atomic.AtomicLong
    import kotlinx.coroutines.flow.MutableStateFlow
    import kotlinx.coroutines.flow.StateFlow
    import kotlinx.coroutines.flow.asStateFlow
    import javax.inject.Inject

    /**
     * 气泡列表管理子组件（per D-23-01, D-23-03, ARCH-10）。
     * 管理气泡列表、去重、线程安全 ID 生成（AtomicLong 替代 Int）。
     * 实现 WS优先/REST降级数据源策略（addFromRest），消除双写 UI 闪烁。
     *
     * 架构选择：使用 @Inject constructor — 同 ConnectionOrchestrator 说明。
     */
    class BubbleMerger @Inject constructor() {
        // D-23-03: AtomicLong 替代 Int，线程安全
        private val bubbleIdCounter = AtomicLong(0L)

        private val _bubbles = MutableStateFlow<List<ChatBubble>>(emptyList())
        val bubbles: StateFlow<List<ChatBubble>> = _bubbles.asStateFlow()

        /** 生成线程安全的气泡 ID */
        fun nextBubbleId(): Long = bubbleIdCounter.incrementAndGet()

        /** 添加用户气泡 */
        fun addUserBubble(text: String, commandType: CommandType) {
            val bubble = ChatBubble(
                id = nextBubbleId(),
                text = text,
                isUser = true,
                commandType = commandType,
                timestamp = System.currentTimeMillis()
            )
            _bubbles.value = _bubbles.value + bubble
        }

        /** 添加 AI 气泡（去重：同一条 WS 事件不重复添加） */
        fun addAiBubble(text: String, speaker: String? = null, dedupeKey: String? = null) {
            val current = _bubbles.value
            if (dedupeKey != null && current.any { it.dedupeKey == dedupeKey }) return
            val bubble = ChatBubble(
                id = nextBubbleId(),
                text = text,
                isUser = false,
                speaker = speaker,
                dedupeKey = dedupeKey,
                timestamp = System.currentTimeMillis()
            )
            _bubbles.value = current + bubble
        }

        /** ARCH-10: WS优先/REST降级数据源策略。
         * 当 WS 连接时（isWsConnected=true），REST 更新被静默丢弃（WS 是唯一真相源）。
         * 当 WS 断开时（isWsConnected=false），REST 更新被接受（降级模式）。
         * 主 VM 在处理 REST 回调时调用此方法，传入 connectionOrchestrator.isWsConnected.value。
         */
        fun addFromRest(bubbles: List<ChatBubble>, isWsConnected: Boolean): List<ChatBubble> {
            if (isWsConnected) return emptyList()  // WS 连接时拒绝 REST 气泡 — SingleSourceOfTruth
            // WS 断开降级模式：接受 REST 气泡，去重合并
            val current = _bubbles.value
            val newBubbles = bubbles.filterNot { restBubble ->
                restBubble.dedupeKey != null && current.any { it.dedupeKey == restBubble.dedupeKey }
            }
            if (newBubbles.isNotEmpty()) {
                _bubbles.value = current + newBubbles
            }
            return newBubbles
        }

        /** 添加错误气泡 */
        fun addErrorBubble(message: String) {
            val bubble = ChatBubble(
                id = nextBubbleId(),
                text = message,
                isUser = false,
                isError = true,
                timestamp = System.currentTimeMillis()
            )
            _bubbles.value = _bubbles.value + bubble
        }

        /** 清空气泡 */
        fun clear() {
            _bubbles.value = emptyList()
            bubbleIdCounter.set(0L)
        }

        /** D-23-04: 主 VM onCleared 时调用 */
        fun cleanup() {
            clear()
        }
    }

    data class ChatBubble(
        val id: Long,
        val text: String,
        val isUser: Boolean,
        val commandType: CommandType? = null,
        val speaker: String? = null,
        val isError: Boolean = false,
        val dedupeKey: String? = null,
        val timestamp: Long = 0L
    )
    ```

    ```kotlin
    import kotlinx.coroutines.flow.MutableSharedFlow

    import javax.inject.Inject

    /**
     * 命令路由子组件（per D-23-01, D-23-02）。
     * 处理命令分发、群聊消息、mention 路由。
     */
    class CommandRouter @Inject constructor(
        private val dramaRepository: DramaRepository
    ) {
        sealed class CommandAction {
            data class SendToServer(val commandType: CommandType, val text: String, val result: Result<CommandResponseDto>) : CommandAction()
            data class LocalAction(val commandType: CommandType, val text: String) : CommandAction()
            data class Error(val message: String) : CommandAction()
        }

        private val _actions = MutableSharedFlow<CommandAction>(extraBufferCapacity = 32)
        val actions: SharedFlow<CommandAction> = _actions

        /** 判断命令类型并路由 */
        fun route(text: String): CommandType {
            return CommandType.fromInput(text)
        }

        /** 判断是否为 action 命令（需要 isTyping + isProcessing） */
        fun isActionCommand(commandType: CommandType): Boolean {
            return commandType in listOf(
                CommandType.ACTION, CommandType.NEXT,
                CommandType.STEER, CommandType.AUTO, CommandType.STORM
            )
        }

        /** 判断是否为剧情变更命令 */
        fun isPlotChanging(commandType: CommandType): Boolean {
            return commandType in listOf(
                CommandType.NEXT, CommandType.ACTION, CommandType.SPEAK, CommandType.FREE_TEXT,
                CommandType.STEER, CommandType.AUTO, CommandType.STORM
            )
        }

        /** 判断是否为本地命令（save/load/list/delete/export） */
        fun isLocalCommand(commandType: CommandType): Boolean {
            return commandType in listOf(CommandType.SAVE, CommandType.LOAD, CommandType.LIST, CommandType.DELETE)
        }

        /** D-23-04: 主 VM onCleared 时调用 */
        fun cleanup() {
            // CommandRouter 无持久状态，无需特殊清理
        }
    }
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| 子组件 SharedFlow → 主 VM | 内部信任边界，事件上报无需验证（同进程） |
| WS 连接 → 远程服务器 | 网络信任边界，WS 消息需校验格式 |
| R8 混淆 → 反编译 | 混淆增加逆向难度但不等于安全 |
| acquire/release → WS 生命周期 | 引用计数防止过早断开或泄漏 |
| BubbleMerger.addFromRest → ConnectionOrchestrator.isWsConnected | 数据源策略依赖连接状态查询，状态不一致可能导致数据丢失或重复 |

|-----------|----------|-----------|-------------|-----------------|
| T-23-01 | T (Tampering) | BubbleMerger dedupeKey | accept | 去重键由服务端事件 ID 生成，客户端不可伪造 |
| T-23-02 | D (Denial of Service) | acquire/release 无上限 | mitigate | AtomicInteger refCount + 上限检查，防止无限 acquire 导致连接永不释放 |
