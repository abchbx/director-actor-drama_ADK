    创建 `data/local/ServerPreferences.kt`：
    ```kotlin
    package com.drama.app.data.local

    import android.content.Context
    import androidx.datastore.preferences.core.edit
    import androidx.datastore.preferences.core.stringPreferencesKey
    import androidx.datastore.preferences.preferencesDataStore
    import dagger.hilt.android.qualifiers.ApplicationContext
    import kotlinx.coroutines.flow.Flow
    import kotlinx.coroutines.flow.first
    import kotlinx.coroutines.flow.map
    import kotlinx.coroutines.runBlocking
    import javax.inject.Inject
    import javax.inject.Singleton

    /**
     * BaseUrl 配置管理（per D-23-10）。
     * 内存缓存 + DataStore 持久化，切换立即生效。
     */
    @Singleton
    class ServerPreferences @Inject constructor(
        @ApplicationContext private val context: Context
    ) {
        private val Context.dataStore by preferencesDataStore(name = "server_prefs")

        companion object {
            private val KEY_BASE_URL = stringPreferencesKey("base_url")
            private val KEY_WS_URL = stringPreferencesKey("ws_url")
        }

        // 内存缓存，避免每次请求都读 DataStore
        @Volatile
        private var cachedBaseUrl: String? = null

        @Volatile
        private var cachedWsUrl: String? = null

        /** 获取当前 BaseUrl（内存缓存优先，per D-23-10） */
        fun currentBaseUrl(): String {
            return cachedBaseUrl ?: runBlocking { loadBaseUrlFromDataStore() }
        }

        /** 设置 BaseUrl（更新内存缓存 + 持久化） */
        suspend fun setBaseUrl(url: String) {
            cachedBaseUrl = url
            context.dataStore.edit { prefs ->
                prefs[KEY_BASE_URL] = url
            }
        }

        /** 获取当前 WS URL */
        fun currentWsUrl(): String {
            return cachedWsUrl ?: runBlocking { loadWsUrlFromDataStore() }
        }

        /** 设置 WS URL */
        suspend fun setWsUrl(url: String) {
            cachedWsUrl = url
            context.dataStore.edit { prefs ->
                prefs[KEY_WS_URL] = url
            }
        }

        private suspend fun loadBaseUrlFromDataStore(): String {
            return context.dataStore.data.map { it[KEY_BASE_URL] }.first()
                ?: "https://api.drama.app"  // 默认生产 URL
                .also { cachedBaseUrl = it }
        }

        private suspend fun loadWsUrlFromDataStore(): String {
            return context.dataStore.data.map { it[KEY_WS_URL] }.first()
                ?: "wss://api.drama.app"  // 默认生产 WS URL
                .also { cachedWsUrl = it }
        }
    }
    ```

    **2. BaseUrlInterceptor.kt — 动态 BaseUrl 切换拦截器（per D-23-10）**

    创建 `data/remote/interceptor/BaseUrlInterceptor.kt`：
    ```kotlin
    package com.drama.app.data.remote.interceptor

    import com.drama.app.data.local.ServerPreferences
    import okhttp3.HttpUrl
    import okhttp3.HttpUrl.Companion.toHttpUrlOrNull
    import okhttp3.Interceptor
    import okhttp3.Request
    import okhttp3.Response
    import javax.inject.Inject

    /**
     * 动态 BaseUrl 切换拦截器（per D-23-10）。
     * 从 ServerPreferences 读取当前 BaseUrl，替换请求 URL。
     * 支持运行时切换，无需重启应用。
     */
    class BaseUrlInterceptor @Inject constructor(
        private val serverPreferences: ServerPreferences
    ) : Interceptor {

        override fun intercept(chain: Interceptor.Chain): Response {
            val originalRequest = chain.request()
            val currentBaseUrl = serverPreferences.currentBaseUrl()
            val newBaseUrl = currentBaseUrl.toHttpUrlOrNull() ?: return chain.proceed(originalRequest)

            val newUrl = originalRequest.url.newBuilder()
                .scheme(newBaseUrl.scheme)
                .host(newBaseUrl.host)
                .port(newBaseUrl.port)
                .build()

            val newRequest = originalRequest.newBuilder()
                .url(newUrl)
                .build()

            return chain.proceed(newRequest)
        }
    }
    ```

    fun connect(dramaId: String, token: String, scope: CoroutineScope) {
        disconnect()
        connectGeneration++
        val thisGeneration = connectGeneration

        // D-23-06: acquire 引用计数
        webSocketManager.acquire()

        connectJob = scope.launch {
            try {
                // D-23-16: WS 优先策略
                webSocketManager.connect(dramaId, token)
                if (thisGeneration == connectGeneration) {
                    _isWsConnected.value = true
                    _events.tryEmit(ConnectionEvent.Connected(dramaId))
                }
            } catch (e: Exception) {
                if (thisGeneration == connectGeneration) {
                    // D-23-16: WS 失败 → REST 降级
                    _isWsConnected.value = false
                    Log.w(TAG, "WS 连接失败，降级为 REST 轮询: ${e.message}")
                    _events.tryEmit(ConnectionEvent.Error("WS 连接失败，降级为 REST 轮询: ${e.message}"))
                    startReplyPolling(dramaId, scope)
                }
            }
        }
    }
