         ```kotlin
         fun exportAndShare() {
             viewModelScope.launch {
                 dramaRepository.exportDrama()
                     .onSuccess { response ->
                         if (response.content.isBlank()) {
                             _events.emit(DramaDetailEvent.ShowSnackbar("导出内容为空"))
                             return@onSuccess
                         }
                         // Write to cache file
                         val timestamp = java.text.SimpleDateFormat("yyyyMMdd_HHmmss", java.util.Locale.US).format(java.util.Date())
                         val fileName = "drama_${timestamp}.md"
                         val file = java.io.File(application.cacheDir, fileName)
                         file.writeText(response.content)

                         // Share via FileProvider
                         val uri = androidx.core.content.FileProvider.getUriForFile(
                             application,
                             "${application.packageName}.fileprovider",
                             file,
                         )
                         val shareIntent = android.content.Intent(android.content.Intent.ACTION_SEND).apply {
                             type = "text/markdown"
                             putExtra(android.content.Intent.EXTRA_STREAM, uri)
                             addFlags(android.content.Intent.FLAG_GRANT_READ_URI_PERMISSION)
                         }
                         _events.emit(DramaDetailEvent.ShareScript(shareIntent))
                     }
                     .onFailure { e ->
                         _events.emit(DramaDetailEvent.ShowSnackbar("导出失败：${e.message}"))
                     }
             }
         }
         ```
